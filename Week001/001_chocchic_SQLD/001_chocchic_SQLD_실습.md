```sql  
-- 3월에 태어난 여성 회원 목록 출력하기
SELECT MEMBER_ID,	MEMBER_NAME,	GENDER,	TO_CHAR(DATE_OF_BIRTH, 'YYYY-MM-DD')
FROM MEMBER_PROFILE 
WHERE TO_CHAR(DATE_OF_BIRTH,'MM') = '03'
AND TLNO IS NOT NULL
AND GENDER = 'W'
ORDER BY MEMBER_ID ASC
```  

```sql  
-- 그룹별 조건에 맞는 식당 목록 출력하기
SELECT P.MEMBER_NAME, R.REVIEW_TEXT, TO_CHAR(R.REVIEW_DATE, 'YYYY-MM-DD') AS REVIEW_DATE
FROM MEMBER_PROFILE P, REST_REVIEW R
WHERE P.MEMBER_ID = R.MEMBER_ID
AND P.MEMBER_ID IN (SELECT MEMBER_ID
FROM (SELECT * FROM (SELECT MEMBER_ID,COUNT(MEMBER_ID) AS COUNT FROM REST_REVIEW group by member_id ORDER BY COUNT DESC) WHERE ROWNUM = 1)
)
ORDER BY REVIEW_DATE ASC, REVIEW_TEXT ASC

/*KEEP 함수 사용 ... 마지막 조건에서 P.MEMBER_ID = (SELECT~) 해도 됨*/
SELECT P.MEMBER_NAME, R.REVIEW_TEXT, TO_CHAR(R.REVIEW_DATE, 'YYYY-MM-DD') AS REVIEW_DATE
FROM MEMBER_PROFILE P, REST_REVIEW R
WHERE P.MEMBER_ID = R.MEMBER_ID
AND P.MEMBER_ID IN (SELECT MAX(MEMBER_ID) KEEP(DENSE_RANK LAST order by COUNT(MEMBER_ID))
FROM REST_REVIEW group by member_id)
ORDER BY REVIEW_DATE ASC, REVIEW_TEXT ASC
```  

```sql  
-- 특정 기간동안 대여 가능한 자동차들의 대여비용 구하기
SELECT * FROM (
SELECT R.CAR_ID, P.CAR_TYPE, ROUND(R.DAILY_FEE * 0.3 * (100-P.DISCOUNT_RATE)) FEE
FROM CAR_RENTAL_COMPANY_CAR R, CAR_RENTAL_COMPANY_DISCOUNT_PLAN P
WHERE R.CAR_TYPE = P.CAR_TYPE
AND R.CAR_ID NOT IN (SELECT CAR_ID
                    FROM CAR_RENTAL_COMPANY_RENTAL_HISTORY
                    WHERE (TO_CHAR(START_DATE, 'YYYYMM') <= '202211' 
                    AND TO_CHAR(END_DATE, 'YYYYMM') >= '202211'))
AND R.CAR_TYPE IN ('SUV', '세단')
AND P.DURATION_TYPE = '30일 이상'
ORDER BY FEE DESC, CAR_TYPE, CAR_ID DESC    
)
WHERE FEE BETWEEN 500000 AND 1999999
/*WHERE FEE >= 500000 AND FEE < 2000000*/
```
